# CMakeLists.txt para o visualizador de NES# Definir arquivos fonteset(NES_VISUALIZER_SOURCES    main.cpp    nes_visualizer.cpp)# Definir arquivos de cabeçalhoset(NES_VISUALIZER_HEADERS    nes_visualizer.hpp)# Adicionar executáveladd_executable(nes_visualizer ${NES_VISUALIZER_SOURCES} ${NES_VISUALIZER_HEADERS})# Configurar flags de compilaçãotarget_compile_features(nes_visualizer PRIVATE cxx_std_17)# Adicionar diretórios de inclusãotarget_include_directories(nes_visualizer PRIVATE    ${CMAKE_SOURCE_DIR}/src    ${SDL2_INCLUDE_DIRS})# Linkar com bibliotecastarget_link_libraries(nes_visualizer    mega_emu_nes    mega_emu_core    ${SDL2_LIBRARIES})# Copiar DLLs do SDL2 para o diretório de saída (Windows)if(WIN32 AND EXISTS ${SDL2_DLLS})    add_custom_command(        TARGET nes_visualizer POST_BUILD        COMMAND ${CMAKE_COMMAND} -E copy_if_different        ${SDL2_DLLS}        $<TARGET_FILE_DIR:nes_visualizer>    )endif()# Instalaçãoinstall(TARGETS nes_visualizer    RUNTIME DESTINATION bin)# Adicionar target para execução fáciladd_custom_target(run_nes_visualizer    COMMAND nes_visualizer    DEPENDS nes_visualizer    WORKING_DIRECTORY ${CMAKE_BINARY_DIR}/bin)# Adicionar informações para documentaçãoset_target_properties(nes_visualizer PROPERTIES    FOLDER "Tools"    PROJECT_LABEL "NES Visualizer"    DESCRIPTION "Visualizador para emulador de NES")
