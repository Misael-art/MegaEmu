# CMake para o NES (Nintendo Entertainment System)cmake_minimum_required(VERSION 3.10)# Definição do projeto NESproject(NES)# Configuração de diretóriosset(NES_SOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR})set(NES_INCLUDE_DIR ${CMAKE_CURRENT_SOURCE_DIR})# Organização dos arquivos fonteset(NES_CORE_SOURCES    ${NES_SOURCE_DIR}/nes.cpp    ${NES_SOURCE_DIR}/nes_integration.cpp)set(NES_CORE_HEADERS    ${NES_SOURCE_DIR}/nes.hpp    ${NES_SOURCE_DIR}/nes_integration.hpp)# Adiciona os subdiretórios dos componentesadd_subdirectory(cpu)add_subdirectory(ppu)add_subdirectory(apu)add_subdirectory(cartridge)add_subdirectory(memory)add_subdirectory(input)# Cria diretório de testes separadoif(BUILD_TESTS)    add_subdirectory(tests)endif()# Biblioteca principal do NESadd_library(nes_core STATIC    ${NES_CORE_SOURCES}    ${NES_CORE_HEADERS})# Configuração de includestarget_include_directories(nes_core PUBLIC    ${NES_INCLUDE_DIR}    ${NES_INCLUDE_DIR}/cpu    ${NES_INCLUDE_DIR}/ppu    ${NES_INCLUDE_DIR}/apu    ${NES_INCLUDE_DIR}/cartridge    ${NES_INCLUDE_DIR}/memory    ${NES_INCLUDE_DIR}/input)# Vincula as bibliotecas dos componentestarget_link_libraries(nes_core    nes_cpu    nes_ppu    nes_apu    nes_cartridge    nes_memory    nes_input)# Configurações de compilaçãoset_target_properties(nes_core PROPERTIES    CXX_STANDARD 17    CXX_STANDARD_REQUIRED ON)# Exporta a bibliotecaexport(TARGETS nes_core FILE NESConfig.cmake)
